SELECT 
    Группы.номер_группы AS номер_группы,
    Преподаватели.ФИО AS преподаватель,
    Предметы.время_ AS время_занятия
FROM
    Группы
JOIN
    Предметы ON Группы.группа_id = Предметы.группа_id
JOIN
    Преподаватели ON Предметы.преподаватель_id = Преподаватели.преподаватель_id
WHERE
    Предметы.время_ >= '2022-01-01' AND Предметы.время_ <= '2022-04-01';

SELECT ФИО, номер_группы, время_поступления, время_отчисления
FROM Студенты
JOIN Активность ON Студенты.студент_id = Активность.студент_id
JOIN Группы ON Активность.группа_id = Группы.группа_id
WHERE ФИО = 'Иванов Иван Иванович';

SELECT Предметы.дисциплина_id, Оценки.оценка
FROM Оценки
JOIN Предметы ON Оценки.предмет_id = Предметы.предмет_id
WHERE Оценки.студент_id = '1';

SELECT Группы.номер_группы, Дисциплины.название_дисциплины, AVG(Оценки.оценка) AS средний_балл
FROM Группы
JOIN Предметы ON Группы.группа_id = Предметы.группа_id
JOIN Дисциплины ON Предметы.дисциплина_id = Дисциплины.дисциплина_id
JOIN Оценки ON Предметы.предмет_id = Оценки.предмет_id
WHERE Дисциплины.дисциплина_id = 1
GROUP BY Группы.группа_id, Дисциплины.дисциплина_id;
